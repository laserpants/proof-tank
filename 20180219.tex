\begin{theorem} 
  If $F$ is a field, and $F$ has infinitely many elements, then for any two polynomials 
  $$a(x) = a_0 + a_1x + a_2x^2 + \dots$$ and 
  $$b(x) = b_0 + b_1x + b_2x^2 + \dots$$ 
  in $F[x]$.
  $$
    a(x) = b(x) \iff a_0 = b_0, a_1 = b_1, \dots
  $$
\end{theorem}
